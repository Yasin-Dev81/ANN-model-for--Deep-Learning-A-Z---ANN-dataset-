input_layer_shape = X_train.shape[1]
print("input layer shape:", input_layer_shape)

# Input layer
input_layer = Input(shape=(input_layer_shape,), name="input_layer")

# Layer 1
Layer_1 = Dense(15, activation="relu", name="Layer_1", kernel_regularizer=l2(0.001))(input_layer)
Layer_1_bn = BatchNormalization(name="Layer_1_bn")(Layer_1)
Layer_1_d = Dropout(0.2, name="Layer_1_d")(Layer_1_bn)

# Layer 2
Layer_2 = Dense(15, activation="relu", name="Layer_2")(Layer_1_d)
Layer_2_bn = BatchNormalization(name="Layer_2_bn")(Layer_2)
Layer_2_d = Dropout(0.3, name="Layer_2_d")(Layer_2_bn)

# Layer 3
Layer_3 = Dense(10, activation="relu", name="Layer_3")(Layer_2_d)
Layer_3_bn = BatchNormalization(name="Layer_3_bn")(Layer_3)
Layer_3_d = Dropout(0.3, name="Layer_3_d")(Layer_3_bn)

# Layer 4
Layer_4 = Dense(10, activation="relu", name="Layer_4", kernel_regularizer=l2(0.001))(Layer_3_d)
Layer_4_bn = BatchNormalization(name="Layer_4_bn")(Layer_4)
Layer_4_d = Dropout(0.2, name="Layer_4_d")(Layer_4_bn)

# Output layer
y_output = Dense(1, activation="sigmoid", name="y_output")(Layer_4_d)

# Build the model
model = Model(inputs=input_layer, outputs=[y_output])
